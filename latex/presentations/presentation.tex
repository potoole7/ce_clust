\begin{frame}{Problem} 
  \begin{itemize}
    \item Often want to estimate
      \[
        \mathbb{P}(X > x, Y > y) = \textcolor{red}{\mathbb{P}(Y > y \mid X > x)}\textcolor{blue}{\mathbb{P}(X > x)}
      \]
      for large $x, y$
    \item ``concomitant''/concurrent extreme events for random vector $\boldsymbol{X}$ often particularly devastating
    \item Goal: \textbf{identify trends by clustering sites with similar \textcolor{red}{tail dependence}}
  \end{itemize}
\end{frame}

\section{Dependence Modelling}

\begin{frame}[fragile]{Asymptotic dependence}
  % reserve space for your citation block
  \newlength{\TopBoxHeight}
  \setlength{\TopBoxHeight}{\dimexpr\textheight-3cm\relax}

  \vbox to \TopBoxHeight{
    \vfill
    \begin{center}
      \begin{itemize}
        % \item Asymptotic dependence is a measure of the strength of dependence between two random variables in the tail of their distribution.
        \item Coefficient of extremal dependence $\chi\in[0,1]$, 
              \[
                \chi = \lim_{u\to1}\mathbb{P}\bigl[F_1(X_1)>u\mid F_2(X_2)>u\bigr]
              \]
        \item (Increasingly strong) asymptotic dependence for $\chi > 0$.
        \item However, $\chi$ only gives summary; inference requires \textbf{dependence model}.
      \end{itemize}
    \end{center}
    \vfill
  }

\begin{frame}[t]{Conditional extremes}
% Reduces multivariate model to simple univariate models!
\begin{itemize}
  \item \textbf{Heteroskedastic regression dependence model}: 
  \begin{equation*}
    \bigl(Y \mid X = x\bigr) = \alpha x + x^{\beta} Z, \text{ for } x > u
  \end{equation*}
  \vspace{-5mm}
  % with
  \begin{itemize}
    \item slope parameter $\alpha$ for $Y$ given large $X$,
    \item $\beta \in (-\infty, 1]$ controls ``spread'' induced by large $X$. 
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t, noframenumbering]{Conditional extremes}
% Reduces multivariate model to simple univariate models!
\begin{itemize}
  \item <1-> \textbf{Heteroskedastic regression dependence model}: 
  \begin{equation*}
      \bigl(\bm{Y}_{-i} \mid Y_i = y \bigr) = \bm{\alpha}_{\mid i}\bm{y} + \bm{y}^{\bm{\beta}_{\mid i}}\bm{Z}_{\mid i}, \text{ for } y > u_{i}
  \end{equation*}
  \vspace{-5mm}
  % with
  \begin{itemize}
    \item <1-> slope parameter $\alpha_{j \mid i} \in [-1, 1]$  for $Y_j$ given large $Y_i$,
    % \item <1-> ``spread'' parameter $\beta_{j \mid i} \in (-\infty, 1]$ controls stochasticity of relationship between $Y_j$ and large $Y_i$. 
    \item $\beta_{j \mid i} \in (-\infty, 1]$ controls ``spread'' induced by large $Y_i$. 
  \end{itemize}
  \vspace{2mm}
  \item <2-> Key assumptions:
  \begin{itemize}
    \item <2-> Residuals $\bm{Z}_{\mid i} \sim N(\bm{\mu}_{\mid i}, \bm{\Sigma}_{\mid i})$ 
    \item <2-> $\bm{Z}_{\mid i}, Y_i$ conditionally independent for large $Y_i$
  \end{itemize}
  \vspace{2mm}
\end{itemize}

\begin{frame}{Inference}
    Inference assumes conditional distribution follows a multivariate Normal ($\mathrm{MVN}$) distribution:
      \begin{equation*}
      \hspace*{-1em} % shift left to fit in slide
      \bigl(\boldsymbol{Y}_{-i, s} \mid Y_{i, s} = y\bigr) \sim \mathrm{MVN}_{d-1}\left(\mvmean{s}, \mvcov{s}\right),
      \end{equation*}
      where $y > u_{i, s}$. \\
      \vspace{1em}
      $\implies$ \textbf{dependence structures at different sites can be compared using their $\mathrm{MVN}$ distributions}
\end{frame}

\begin{frame}{Kullback-Leibler divergence}
  \begin{itemize} \setlength{\itemsep}{2pt}
  \item $\KL{}$ between two continuous $(d-1)$-variate distribution functions $H$ and $H^*$ is 
    \begin{equation*}
      \KL{(H \Vert H^*)} = \int_{\mathbb{R}^{d-1}} h(\symbf{y}) \log\left(\frac{h(\symbf{y})}{h^*(\symbf{y})}\right) \mathrm{d}\symbf{y},
    \end{equation*}
    where $h, h^*$ are continuous density functions for $H, H^*$
  \item <2-> $\KL{}$ has closed form when $H$ and $H^*$ are $\mathrm{MVN}$
  % \item <3-> However, asymmetric and can ``blow up'' if supports poorly overlap
  % \item <4-> $\KL{(H \Vert H^*)} + \KL{(H^* \Vert H)}$ symmetric but still unstable
  \item <3-> $\KL{(H \Vert H^*)} + \KL{(H^* \Vert H)}$ symmetric but can be unstable
  \end{itemize}
\end{frame}

\begin{frame}{Skew-geometric Jensen-Shannon divergence}
\begin{itemize}%\setlength{\itemsep}{2pt}  % smaller space (default ~6pt)
  % \item $\JSGa{}$ between two continuous $(d-1)$-variate distribution functions $H$ and $H^*$ is 
  \item $\JSGa{}$ between $H, H^*$ is

  \begin{align*}
  \JSGa{(H \Vert H^*)}

\begin{frame}{Conditional $\JSGa{}$}
  \begin{itemize}
    % \item <1-> \emph{Conditional} $\JSGa{}$ between two sites $s$ and $s^*$ given conditioning variable $i$:
    \item <1-> $\JSGa{}$ for sites $s, s^*$ \emph{conditional} on variable $i$:
      \[
        \cJSGa{s, s^*}(y) = \JSGa{}\left(H_{i, s}(y) \Vert H_{i, s^*}(y)\right),\qquad y > u_i,
      \]
      where $H_{i, s}(y), H_{i, s^*}(y)$ are conditional $\mathrm{MVN}$s at $s, s^*$
      % use the same u_i across all sites to be comparable
    \item <2-> $\cJSGa{}$ depends on choice of $y > u_i$ $\ldots$
    \end{itemize}
\end{frame}

\begin{frame}{Expected $\cJSGa{s, s^*}$}
  % \begin{itemize}
    % \item <1-> \emph{Expected} $\cJSGa{}$:
    \emph{Expected} $\cJSGa{s, s^*}$:
      % Symmetric (gives same result if we swap s and s*)
      \begin{align*}
  \eJSGa{s, s^*} &= \mathbb{E}\left(\cJSGa{s, s^*}(Y_{i,s}) \mid Y_{i,s} > u_i\right) \\
                 % &= \mathbb{E}\left(\cJSGa{s^*, s}(Y_{i,s^*}) \mid Y_{i,s^*} > u_i\right) \\
                 &= \int_{u_i}^{\infty} \cJSGa{s, s^*}(y) h(y \mid Y_{i,s} > u_i) \mathrm{d}y
\end{align*}
  where $h(y \mid Y_{i,s} > u_i)$ is on standard Laplace scale.
  % \end{itemize}
\end{frame}

\begin{frame}{Dissimilarity Matrix}
  \begin{itemize}
    \item <1-> For each conditioning variable $i \in \{1, \ldots, d\}$, compute pairwise $\eJSGai{}$ dissimilarities between all $D$ sites:
    \begin{equation*} 
      M^{(i)} \coloneq \left(\eJSGa{s, s^*} \right)_{s, s^* = \{1, \ldots, D\}}
    \end{equation*}
  \item Can aggregate over all conditioning variables to get overall dissimilarity matrix:
    \begin{equation*}
      M = \frac{1}{d} \sum_{i=1}^{d} M^{(i)}
    \end{equation*}
  \end{itemize}
\end{frame}

\begin{frame}{Clustering}

\begin{columns}[T,totalwidth=\textwidth]
  \column{0.55\textwidth}
  \begin{itemize}\setlength{\itemsep}{2pt}
    \item Use \textbf{k-medoids} to cluster over $M$ (or $M^{(i)}$) % dissimilarity matrix
    \item \textbf{Elbow plot}: $k$ chosen using \textbf{Total Within Group Sum of Squares}:
      \[
        \mathrm{TWGSS}(k)
          = \sum_{c=1}^{k}
             \sum_{s \in \mathcal{C}_c}
               \eJSGa{s, \kappa_c},
      \]
      where $\mathcal{C}_c$, $\kappa_c$ are cluster $c$ and its medoid
  \end{itemize}

\section{Discussion}

\begin{frame}{Discussion}
  \begin{itemize}
    \item <1-> \textbf{Conclusion}:
      \begin{itemize}
        \item Principled \& effective clustering framework for CE models
        \item Simulations \& application show clustering can group sites with similar tail dependence, which may aid interpretation
      \end{itemize} 
    \vspace{2mm}
    \item <2-> \textbf{Limitations}:
      \begin{itemize}
        \item \textbf{CE}: Gaussian assumption for $\bm{Z}_{\mid i}$
        \item \textbf{Clustering}: Uncertainty in CE fits is ignored in clustering
      \end{itemize} 
  \end{itemize} 
\end{frame}

    \begin{align*}
    \JSGa{}(H \Vert H^*)
      &= \tfrac12 \Bigl[
          (1-\lambda)\,\symbf{m}^T \Omega^{-1} \symbf{m}
          + \lambda (\symbf{m}^*)^T (\Omega^*)^{-1} \symbf{m}^*
    \\[-0.2em] &\quad
          - \symbf{m}_{\lambda}^T \Omega_{\lambda}^{-1} \symbf{m}_{\lambda}
          + \log\!\left(
              \frac{
                \abs{\Omega}^{1-\lambda} \abs{\Omega^*}^{\lambda}
              }{
                \abs{\Omega_{\lambda}}
              }
            \right)
        \Bigr], \\[0.3em]
    \symbf{m}_{\lambda}
      &= \Omega_{\lambda}\!\left[
          (1-\lambda)\Omega^{-1}\symbf{m}
          + \lambda(\Omega^*)^{-1}\symbf{m}^*
        \right], \\
    \Omega_{\lambda}
      &= \left[
          (1-\lambda)\Omega^{-1}
          + \lambda(\Omega^*)^{-1}
        \right]^{-1}. 
    \end{align*}
  \item $\Omega_{\lambda}$ is matrix barycentre of $\Omega$ and $\Omega^*$ 
  \item $\JSGa{(H \Vert H^*)} \leq \frac{1}{2}\max\{\lambda,\,1-\lambda\}\left[\KL{}(H \Vert H^*) + \KL{}(H \Vert H^*)\right]$
  \end{itemize}
