\section{Dependence Modelling}

\begin{frame}{Asymptotic dependence}
\begin{itemize}
  % \item Asymptotic dependence is a measure of the strength of dependence between two random variables in the tail of their distribution.
  \item Asymptotic dependence measures strength of tail dependence between two random variables % or more accurately their CDFs
  % \item Can occur even in absence of ``bulk'' dependence.
  % \item The coefficient of dependence, $\chi \in [0, 1]$, measures the degree of extremal dependence between $X$ and $Y$ with corresponding CDFs $F_1$ and $F_2$.
  \item Coefficient of dependence, $\chi \in [0, 1]$, measures extremal dependence between $X, Y$ with CDFs $F_X, F_Y$
  % \item Defined as $\chi = \lim_{u \rightarrow 1}{\chi(u)}$, where
  \item $\chi = \lim_{u \rightarrow 1}{\chi(u)}$ where
\[
\chi(u) = 2 - \frac{\log\left(\mathbb{P}(F_X(X) < u, F_Y(Y) < u)\right)}{\log \left(\mathbb{P}(F_1(X) < u)\right)}. % log ratio of joint to marginal CDF as threshold increases
\]
\end{itemize}
\end{frame}

\begin{frame}{Copulas}
  \begin{itemize}

    \item multivariate CDF with Uniform $[0, 1]$ margins:
      \begin{align*}
        C(u_1,\dots,u_d) &= F\bigl(F_1^{-1}(u_1),\dots,F_d^{-1}(u_d)\bigr) \\
                         % &= \mathbb{P}(F_1(X_1) < u_1, \dots, F_d(X_d) < u_d)

    \item \vspace{-8mm} Models marginals and dependence separately, margins can be any distribution.
    \item Choices include:

        \item Gaussian: 
          \[
            C_{\Sigma}(u) = \Phi_{\Sigma}\Bigl(\Phi^{-1}(u_1),\dots,\Phi^{-1}(u_d)\Bigr)
          \]
        \item Student-t:
          \[
            C_{\Sigma,\nu}(u) = t_{\Sigma,\nu}\Bigl(t_\nu^{-1}(u_1),\dots,t_\nu^{-1}(u_d)\Bigr)
          \]
      \end{itemize}
    \item Lacks flexibility \& efficiency of CE, but \textbf{used to easily simulate data}.

\section{Clustering}

\begin{frame}{Clustering}
    \begin{itemize}
      % \item CE very flexible \& conceptually simple, but can be difficult to fit, with large estimation uncertainty and difficult interpretations for parameters.
      \item CE flexible \& conceptually simple, but often difficult to fit \& interpret under large uncertainty. % , with large estimation uncertainty and difficult interpretations for parameters.
      % \item extensions to CE model which seek to overcome shortcomings of ``vanilla'' model, such as spatial and spatio-temporal versions.
      % \item Alternative/additional approach: \textbf{clustering}
      \item Clustering can help with
        \begin{itemize}
          \item \textbf{enhancing interpretability of data},
          \item improving parameter estimation.
        \end{itemize}
      % \item \textbf{How best to cluster for CE model?}
      \item \textbf{Aim}: Find suitable distance metric/measure to estimate dissimilarity matrix across sites based on CE fits, upon which we can perform clustering. 
    \end{itemize}
\end{frame}

\begin{frame}{Kullback-Leibler divergence}
  % The Kullback-Leibler (KL) divergence between two continuous distributions $P$, $Q$ is given by
  Kullback-Leibler (KL) divergence between two distributions $P$, $Q$ is
  \[
    KL(P \mid\mid Q) = \int_{-\infty}^{\infty} p(y) \log\left(\frac{p(y)}{q(y)}\right) dy.
  \]
  \vspace{-5mm}
  \begin{itemize}
  \item For two MVNs, KL has closed form. 
  \item However, KL asymmetric ($KL(P \mid\mid Q) \ne KL(Q \mid\mid P)$) and not ``true'' metric $\implies$ poor choice of dissimilarity measure. 
  \end{itemize}
\end{frame}

\begin{frame}{Jensen-Shannon divergence}
  \begin{itemize}
    \item The Jensen-Shannon divergence given by
      \[
        JS(P \mid\mid Q) = \frac{1}{2} KL(P \mid\mid M) + \frac{1}{2} KL(Q \mid\mid M),
      \]
      % where $M = \frac{1}{2}(P + Q)$ is a mixture distribution.
      where $M$ is two-component mixture of $P, Q$. 
    \item JS is symmetric \& true metric!
    \item However, mixture of two normals is non-normal \\ $\implies$ no closed form for two MVNs. 
  \end{itemize}
\end{frame}

\begin{frame}{skew-geometric Jensen-Shannon divergence}
\begin{itemize}
  \item Introduce the weighted geometric mean $G_{\alpha}(x, y) = x^{\alpha} y^{1 - \alpha}, \alpha \in [0, 1]$
  % \item The weighted product of two members of the same exponential family also comes from this family, so we will have another multivariate normal. 
  \item weighted product of two members of same exponential family comes from \textbf{same family}.
  \item Using $G_{\alpha}$, we get the \textbf{skew-geometric Jensen-Shannon divergence}
    \[
      JS^{G_{\alpha}}(P \mid\mid Q) = \frac{1}{2} KL(P \mid\mid G_{\alpha}(P, Q)) + \frac{1}{2} KL(Q \mid\mid G_{\alpha}(P, Q))
    \]
  % \item Has both closed form for MVNs and is true metric, making it a good candidate as a dissimilarity measure for clustering. 
  \item Has both closed form for MVNs and is true metric $\implies$ found suitable dissimilarity measure!
  \item Sensible choice is $\alpha = 0.5$. %, may adapt data-driven or theoretically motivated approach in the future. 
\end{itemize}
\end{frame}

\begin{frame}{Partitioning Around Mediods (PAM)}
  \begin{itemize}
    % \item Now we have a dissimilarity measure, we can use it to cluster the data.
    \item Require clustering algorithm to cluster data using dissimilarity matrix.
    % \item We use the PAM algorithm, which is a partitioning clustering algorithm that uses the mediod of each cluster as the representative point.
    \item PAM algorithm uses mediod of each cluster as representative point.
    % \item Cluster centroids actual sites, making them more interpretable than the centroids of k-means, especially for spatial and spatio-temporal applications. 
    \item Cluster centroids are actual sites, giving better interpretability and a location to refit at.
    % \item Also, as we are clustering on functional space, advantageous to have actual set of CE parameters rather than some summary as cluster centre.
    % \item Cluster centroids also actual CE fits, want to be careful when clustering in functional space that centroids represent valid models!
  \end{itemize}
\end{frame}

\begin{frame}{Selecting $k$: total within cluster sum of squares \& elbow plot}
  \begin{itemize}
    \item \textbf{TWGSS}:
      \[
        \mathrm{TWGSS}(k)
        % = \sum_{i=1}^{k}\sum_{x \in C_i} d\bigl(x, m_i\bigr), d = JS^{G_{\alpha}}
        = \sum_{i=1}^{k}\sum_{x \in C_i} JS^{G_{\alpha}}\bigl(x, m_i\bigr),
      \]
    \item \textbf{Elbow Plot:} Plot $\mathrm{TWGSS}(k)$ vs.\ $k$, observe curve, choose $k^*$ at ``elbow'' where TWGSS reduction sharply slows
  \end{itemize}

\section{Conclusions}

\begin{frame}{Limitations}
\begin{itemize}
  \item \textbf{CE}: $Z_{\mid i}$ assumed to be normal.
    \begin{itemize}
      \item Requires diagnostic plots to ascertain validity, if not then uncertainty may be underestimated.
    \end{itemize}
  \item Also, especially where data is scarce, fit can be non-identifiable/non-unique (i.e.\ many candidate models valid). 
  \item \textbf{Clustering}: high uncertainty in original CE fits not incorporated in clustering, only MLEs $\hat{\alpha}, \hat{\beta}$ used. 
  \begin{itemize}
    \item While sufficient for interpretation, would require better treatment for improving uncertainty.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Conclusions \& future work}
\begin{itemize}
  \item We provide principled \& effective clustering framework for CE models.
  \item Shown through simulations and real data that clustering can group sites with similar tail dependence, which may help interpretability of CE model, 
  % \item Mainly used for improved interpretability, would like to see if parameter estimation better post-clustering.
  \item Need to compare Ireland clustering to Vignotto. 
  \item Want to apply clustering to more interesting multivariate data, such as Brazil landslide data and air pollution extremes in US cities. 
  \item Explore clustering solutions incorporating uncertainty, and also how well we can do at reducing this uncertainty
\end{itemize}
\end{frame}
